# Screen

sin (fx * 0.5) * [osc 0.01, osc 0.2, osc 0.4] >> add;

sin (fx * 5) * sin (fy * 5)  * [osc 0.01, osc 0.2, osc 0.4] >> blend

# Circles and all

circle [0,0] (0.05 ~~  1   $ osc 0.25) >> blend;
circle [0,0] (0.03 ~~  0.7 $ osc 0.4)  >> blend;
circle [0,0] (0.02 ~~  0.4 $ osc 0.7)  >> blend;

# Flickering circles

sin(log(fx * fy)) * osc((osc ft) * 0.1) >> rgb;
sin (fx * osc 50) * [osc 0.01, osc 0.2, osc 0.4] >> add;

sin (fx * 5) * sin (fy * 5)  * [osc 0.01, osc 0.2, osc 0.4] >> blend;
sin(cos(fx * fy)) * osc((osc ft) * 0.1) >> blend;

circle [0,0] (osc ft * 0.5) * [osc 0.1, osc 0.5, osc 0.8] >> blend; -- Mocing
circle [osc 0.13, osc 0.17] (osc ft * 0.5) * [osc 0.8, osc 0.05, osc 0.1] >> blend;

# Mesh

mesh [0,0, 0.5, 0.5, -0.5, 0.5] 0.01 >> blend

mesh [0,0, osc 0.5, 0.5, -0.5,0.5] 0.01 >> blend

mesh [sin (osc 0.1), cos (osc 0.13), sin (osc 0.17)*0.5, cos (osc 0.07)*0.5, sin (osc 0.23)*0.8, cos (osc 0.19)*0.8] 0.005 >> blend

# Line

hline (osc 0.5) 0.01 >> blend

spin (lfsaw 0.1 + 0.333) $ vline 0 0.005 >> blend;
spin (lfsaw 0.1 + 0.666) $ vline 0 0.005 >> blend;
